-- MySQL

drop database if exists caizimeidb;

create database if not exists caizimeidb;

use caizimeidb;

create table country(
	co_id					int auto_increment not null,
    co_name					nvarchar(10) not null,
    co_eng_name				varchar(50),
    co_rank					tinyint,
    primary key (co_id)
);

create table city(
	ci_id					int auto_increment not null,
    ci_co_id				int not null,
    ci_name					nvarchar(10) not null,
    ci_rank					tinyint,
    primary key (ci_id),
	foreign key (ci_co_id) references country (co_id)
);

create table region(
	r_id					int auto_increment not null,
    r_ci_id					int not null,
    r_name					nvarchar(10) not null,
    r_zipcode				varchar(10),
    r_rank					tinyint,
    primary key (r_id),
    foreign key (r_ci_id) references city (ci_id)
);

create table clinic(
	c_id					int auto_increment not null,
    c_name					nvarchar(20) not null,
    c_eng_name				varchar(50),
    c_localphone			varchar(20),
    c_r_id					int not null,
    c_address				nvarchar(50) not null,
    c_latitude				decimal(10,6),
    c_longitude				decimal(10,6),
    c_url					varchar(100),
    c_insert_time			datetime not null,
    c_update_time			datetime,
    c_status				tinyint,
    c_status_time			datetime,
    primary key (c_id),
    foreign key (c_r_id) references region (r_id)
);

create table administrator(
	a_id					int auto_increment not null,
    a_username				varchar(50) not null,
    a_password				varchar(50) not null,
    a_salt					varchar(50) not null,
    a_lastname				nvarchar(20),
    a_firstname				nvarchar(20),
    a_eng_name				varchar(50),
    a_email					varchar(50) not null,
    a_mobilephone			varchar(20),
    a_signup_time			timestamp default current_timestamp not null,
    a_signin_number			int,
    a_signin_ip				varchar(20),
    a_signin_time			datetime,
    a_update_pass_time		datetime,
    a_update_info_time		datetime,
    primary key (a_id)
);

create table administrator_log(
	al_id					int auto_increment not null,
    al_insert_time			timestamp default current_timestamp not null,
    al_a_id					int not null,
    al_operation			nvarchar(50),
    primary key (al_id),
    foreign key (al_a_id) references administrator (a_id)
);

create table company(
	com_id					int auto_increment not null,
    com_name				nvarchar(20) not null,
    com_localphone			varchar(20),
    com_insert_time			datetime not null,
    com_status				tinyint,
    com_status_time			datetime,
    primary key (com_id)
);

create table employee(
    e_id					int auto_increment not null,
    e_username				varchar(50) not null,
    e_password				varchar(50) not null,
    e_salt					varchar(50) not null,
    e_lastname				nvarchar(20),
    e_firstname				nvarchar(20),
    e_eng_name				varchar(50),
    e_email					varchar(50) not null,
    e_mobilephone			varchar(20),
	e_signup_time			timestamp default current_timestamp not null,
    e_signin_number			int,
    e_signin_ip				varchar(20),
    e_signin_time			datetime,
    e_update_pass_time		datetime,
    e_update_info_time		datetime,
    e_com_id				int not null,
    primary key (e_id),
	foreign key (e_com_id) references company (com_id)    
);

create table member(
	m_id					int auto_increment not null,
    m_username				varchar(50) not null,
    m_password				varchar(50) not null,
    m_salt					varchar(50) not null,
    m_lastname				nvarchar(20),
    m_firstname				nvarchar(20),
    m_birth					date,
    m_sex					tinyint,
    m_height				double,
    m_weight				double,
    m_localphone			varchar(20),
    m_mobilephone			varchar(20),
    m_zipcode				varchar(10),
    m_country				varchar(10),
    m_city					nvarchar(10),
    m_region				nvarchar(10),
    m_address				nvarchar(50),
    m_signup_time			timestamp default current_timestamp not null,
    m_signin_number			int,
    m_signin_ip				varchar(20),
    m_signin_time			datetime,
    m_update_pass_time		datetime,
    m_update_info_time		datetime,
    m_com_id				int not null,
    primary key (m_id),
	foreign key (m_com_id) references company (com_id)
);

create table purchase(
	p_id					int auto_increment not null,
    p_m_id					int not null,
    p_c_id					int not null,
    p_order_time			timestamp default current_timestamp not null,
    primary key (p_id),
    foreign key (p_m_id) references member (m_id),
    foreign key (p_c_id) references clinic (c_id)
);
